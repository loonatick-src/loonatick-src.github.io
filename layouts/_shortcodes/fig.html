{{- /* Numbered figure with a stable anchor for cross-referencing. Usage: {{< fig key="baseline-bw" src="/images/copyif/baseline.svg" alt="Baseline bandwidth" caption="likwid-bench copy bandwidth." >}} Reference it elsewhere in the same page with: {{< figref "baseline-bw" >}} The per-page counter and key->number map live in .Page.Store, so numbers follow source order. Caveat: a {{< figref >}} that appears *before* its {{< fig >}} in the document cannot know the number yet (content renders top-to-bottom) and will fall back to a plain anchor link. */ -}} {{- $key := .Get "key" -}} {{- if not $key -}} {{- errorf "fig: missing required 'key' parameter (%s)" .Position -}} {{- end -}} {{- $n := add (.Page.Store.Get "figCounter" | default 0) 1 -}} {{- .Page.Store.Set "figCounter" $n -}} {{- $map := .Page.Store.Get "figMap" | default dict -}} {{- $map = merge $map (dict $key $n) -}} {{- .Page.Store.Set "figMap" $map -}}
